167 void _unregister_incident_edges(typename Base::Verts::iterator v_iter) {
168 if constexpr (HasEdgeKey()) {
169 // delete the keys for all edges incident to the deleted vert
170 for (EdgeDir dir : {EdgeDir::Incoming, EdgeDir::Outgoing}) {
171 for (auto e = this->begin_incident_edges({this, v_iter}, dir);
172 e != this->end_edges();
173 ++e)

175 _unregister_edge(e->id());
176 }

424 size_t erase(const VertKey& key) requires (HasVertKey()) {
425 auto i = _vert_ids_by_key.find(key);
426 if (i != _vert_ids_by_key.end()) {
427 auto v_iter = this->_verts.find(i->second);
428 _unregister_incident_edges(v_iter);
429 // delete the key for this vert
430 VertexId v_id = i->second;
431 _vert_ids_by_key.erase(i);
432 _vert_keys_by_id.erase(v_id);
433 // delete the vert
434 _base()->erase(vertex_iterator{this, v_iter});
435 return 1;
436 } else {
437 return 0;
438 }
439 }

799 template <typename... Args>
800 std::pair<edge_iterator, EdgeChange> redirect_edge(
801 const EdgeKey& key,
802 vertex_iterator src,
803 vertex_iterator dst,
804 Args... args)
805 requires (HasEdgeKey())
806 {
807 auto e = this->find_edge(key);
808 if (e != this->end_edges()) {
809 if (e->source_id() == src->id() and e->target_id() == dst) {
810 // edge already exists with this key and endpoints
811 if constexpr (HasVertexValue()) {
812 // replace the existing value
813 e->value() = VertVal{std::forward<Args>(args)...};
814 }
815 return {e, EdgeChange::MODIFIED};
816 } else {
817 // edge exists, but points to the wrong verts
818 this->erase(e);
819 return std::make_pair(
820 this->emplace_directed_edge(
821 key,
822 src,
823 dst,
824 std::forward<Args>(args)...
825 ).first,
826 EdgeChange::REDIRECTED
827 );
828 }
829 } else {
830 // edge does not exist. emplace it.
831 return std::make_pair(
832 this->emplace_directed_edge(key, src, dst, std::forward<Args>(args)...).first,
833 EdgeChange::CREATED
834 );
835 }

860 Args... args)

862 {
