157 void _unregister_incident_edges(typename Base::Verts::iterator v_iter) {
158 if constexpr (HasEdgeKey()) {
159 // delete the keys for all edges incident to the deleted vert
160 for (EdgeDir dir : {EdgeDir::Incoming, EdgeDir::Outgoing}) {
161 for (auto e = this->begin_incident_edges({this, v_iter}, dir);
162 e != this->end_incident_edges();
163 ++e)
164 {
165 _unregister_edge(e->id());
166 }
167 }
168 }

404 size_t erase(const VertKey& key) requires (HasVertKey()) {
405 auto i = _vert_ids_by_key.find(key);
406 if (i != _vert_ids_by_key.end()) {
407 auto v_iter = this->_verts.find(i->second);
408 _unregister_incident_edges(v_iter);
409 // delete the key for this vert
410 VertexId v_id = i->second;
411 _vert_ids_by_key.erase(i);
412 _vert_keys_by_id.erase(v_id);
413 // delete the vert
414 _base()->erase(vertex_iterator{this, v_iter});
415 return 1;
416 } else {
417 return 0;
418 }
419 }
