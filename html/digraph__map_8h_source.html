160 void _unregister_incident_edges(typename Base::Verts::iterator v_iter) {
161 if constexpr (HasEdgeKey()) {
162 // delete the keys for all edges incident to the deleted vert
163 for (EdgeDir dir : {EdgeDir::Incoming, EdgeDir::Outgoing}) {
164 for (auto e = this->begin_incident_edges({this, v_iter}, dir);
165 e != this->end_incident_edges();
166 ++e)
167 {
168 _unregister_edge(e->id());
169 }
170 }

417 size_t erase(const VertKey& key) requires (HasVertKey()) {
418 auto i = _vert_ids_by_key.find(key);
419 if (i != _vert_ids_by_key.end()) {
420 auto v_iter = this->_verts.find(i->second);
421 _unregister_incident_edges(v_iter);
422 // delete the key for this vert
423 VertexId v_id = i->second;
424 _vert_ids_by_key.erase(i);
425 _vert_keys_by_id.erase(v_id);
426 // delete the vert
427 _base()->erase(vertex_iterator{this, v_iter});
428 return 1;
429 } else {
430 return 0;
431 }
432 }

792 template <typename... Args>
793 std::pair<edge_iterator, bool> redirect_edge(
794 const EdgeKey& key,
795 vertex_iterator src,
796 vertex_iterator dst,
797 Args... args)
798 requires (HasEdgeKey())
799 {
800 auto e = this->find_edge(key);
801 if (e != this->end_incident_edges()) {
802 if (e->source_id() == src->id() and e->target_id() == dst) {
803 // edge already exists with this key and endpoints
804 if constexpr (HasVertexValue()) {
805 // replace the existing value
806 e->value() = VertVal{std::forward<Args>(args)...};
807 }
808 return {e, false};
809 } else {
810 // edge exists, but points to the wrong verts
811 this->erase(e);
812 return std::make_pair(
813 this->emplace_directed_edge(
814 key,
815 src,
816 dst,
817 std::forward<Args>(args)...
818 ).first,
819 false
820 );
821 }
822 } else {
823 // edge does not exist. emplace it.
824 return std::make_pair(
825 this->emplace_directed_edge(key, src, dst, std::forward<Args>(args)...).first,
826 true
827 );
828 }
