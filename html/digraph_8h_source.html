124inline size_t hash_combine(size_t h0, size_t h1) {
125 if constexpr (sizeof(size_t) == 8) {
126 /*
127 // this offers a solution which doesn't use
128 // a variant of the shitty boost combine:
129 // https://stackoverflow.com/questions/8513911/
130 // but doesn't justify it.
131 constexpr size_t k = 0x9E3779B97F4A7C15ULL;
132 size_t a = (h0 ^ h1) * k;
133 a ^= (a >> 47);
134 size_t b = (h0 ^ a) * k;
135 b ^= (b >> 47);
136 return b * k;
137 */
138
139 // constant from: https://stackoverflow.com/questions/5889238/
140 h0 ^= h1 + 0x517cc1b727220a95 + (h0 << 12) + (h0 >> 4);
141 return h0;
142 } else if constexpr (sizeof(size_t) == 4) {
143 // boost's hash_combine. everyone uses this even though it's known to be shit.
144 // i am not a cryptographer, though, so I don't expect to do much better myself.
145 // we should make this better, though
146 h0 ^= h1 + 0x9e3779b9U + (h0 << 6) + (h0 >> 2);
147 return h0;
148 } else if constexpr (sizeof(size_t) == 2) {
149 // same as above. ugh
150 h0 ^= h1 + 0x9e37U + (h0 << 3) + (h0 >> 1);
151 return h0;
152 } else {
153 static_assert(sizeof(size_t) > 2, "size_t size not supported for hashing");
154 }
155}

962 // assumes the given edge is definitely in the list.
963 // returns the next edge ID in the list.
964 std::optional<EdgeId> _remove_from_list(
965 vertex_iterator v,
966 EdgeDir dir,
967 typename Edges::iterator edge)
968 {
969 EdgeList& list = v->node().edges[(int) dir];
970 if (list.size == 1) {
971 // removing the only edge
972 list.head = std::nullopt;
973 list.size = 0;
974 return std::nullopt;
975 } else {
976 EdgeList& list = v->node().edges[(int) dir];
977 std::optional<EdgeId> edge_id = edge->first;
978 std::optional<EdgeId> prev_id = std::nullopt;
979 std::optional<EdgeId> cur_id = list.head;
980 auto e_prev = _edges.end();
981 auto e_cur = e_prev;
982 while(cur_id.has_value()) {
983 e_cur = _edges.find(*cur_id);
984 std::optional<EdgeId> next_id = e_cur->second.next[(int) dir];
985 if (cur_id == edge_id) {
986 // found the edge to remove; excise it from the list
987 if (not prev_id) {
988 // removing the head
989 list.head = next_id;
990 } else {
991 // removing an interior or tail edge.
992 // make the previous edge skip the deleted one.
993 e_prev->second.next[(int) dir] = next_id;
994 }
995 list.size -= 1;
996 cur_id = next_id;
997 break;
998 } else {
999 prev_id = cur_id;
1000 cur_id = next_id;
1001 e_prev = e_cur;
1002 }
1003 }
1004 return cur_id;
1005 }

1120 vertex_iterator erase(vertex_iterator v) {
1121 if (v == end_vertices()) {
1122 return end_vertices();
1123 }
1124 for (EdgeDir dir : {EdgeDir::Incoming, EdgeDir::Outgoing}) {
1125 for (auto e = begin_incident_edges(v, dir); e != end_incident_edges(); ) {
1126 // remove the edge from the other vertex's list
1127 const Edge& this_edge = e->edge();
1128 // if we're iterating over this vertex's incoming edges, then this vertex is
1129 // the destination, and the other vertex is the edge's source. Vice versa for
1130 // the other direction.
1131 VertexId other_id = (dir == EdgeDir::Incoming) ? this_edge.v0 : this_edge.v1;
1132 vertex_iterator other = find_vertex(other_id);
1133 // remove the edge from the other vertex's (opposite-direction) list
1134 _remove_from_list(other, ~dir, e.inner_iterator());
1135
1136 // don't bother removing the edge from the current list, because we're
1137 // about to erase it anyway. increment the iterator by following its link
1138 // to the next edge now; after we erase the edge, the iterator will
1139 // be invalid.
1140 auto e_iter = e.inner_iterator();
1141 std::optional<EdgeId> next_id = e_iter->second.next[(int) dir];
1142 // erase the edge
1143 _edges.erase(e_iter);
1144 if (not next_id) break;
1145 e = incident_edge_iterator {this, _edges.find(*next_id), dir};
1146 }
1147 }
1148 return {this, _verts.erase(v.inner_iterator())};
