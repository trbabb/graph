150inline size_t hash_combine(size_t h0, size_t h1) {
151 if constexpr (sizeof(size_t) == 8) {
152 /*
153 // this offers a solution which doesn't use
154 // a variant of the shitty boost combine:
155 // https://stackoverflow.com/questions/8513911/
156 // but doesn't justify it.
157 constexpr size_t k = 0x9E3779B97F4A7C15ULL;
158 size_t a = (h0 ^ h1) * k;
159 a ^= (a >> 47);
160 size_t b = (h0 ^ a) * k;
161 b ^= (b >> 47);
162 return b * k;
163 */
164
165 // constant from: https://stackoverflow.com/questions/5889238/
166 h0 ^= h1 + 0x9e3779b97f4a7c15 + (h0 << 12) + (h0 >> 4);
167 return h0;
168 } else if constexpr (sizeof(size_t) == 4) {
169 // boost's hash_combine. everyone uses this even though it's known to be shit.
170 // i am not a cryptographer, though, so I don't expect to do much better myself.
171 // we should make this better, though
172 h0 ^= h1 + 0x9e3779b9U + (h0 << 6) + (h0 >> 2);
173 return h0;
174 } else if constexpr (sizeof(size_t) == 2) {
175 // same as above. ugh
176 h0 ^= h1 + 0x9e37U + (h0 << 3) + (h0 >> 1);
177 return h0;
178 } else {
179 static_assert(sizeof(size_t) > 2, "size_t size not supported for hashing");
180 }
181}

1114 // assumes the given edge is definitely in the list.
1115 // returns the next edge ID in the list.
1116 // does not erase the edge from the graph.
1117 std::optional<EdgeId> _excise_from_list(
1118 vertex_iterator v,
1119 EdgeDir dir,
1120 typename Edges::iterator edge)
1121 {
1122 EdgeList& list = v->node().edges[(int) dir];
1123 if (list.size == 1) {
1124 // removing the only edge
1125 list.head = std::nullopt;
1126 list.size = 0;
1127 return std::nullopt;
1128 } else {
1129 const EdgeNode& e = edge->second;
1130 std::optional<EdgeId> edge_id = edge->first;
1131 std::optional<EdgeId> next_id = e.links[(int) dir].next;
1132 std::optional<EdgeId> prev_id = e.links[(int) dir].prev;
1133
1134 if (prev_id) {
1135 // previous edge exists and should skip the deleted node
1136 _edges.find(*prev_id)->second.links[(int) dir].next = next_id;
1137 } else {
1138 // the deleted edge is was the head of the list
1139 list.head = next_id;
1140 }
1141
1142 if (next_id) {
1143 // next edge exists and should skip the deleted node
1144 _edges.find(*next_id)->second.links[(int) dir].prev = prev_id;
1145 } else {
1146 // the deleted edge was the tail of the list
1147 list.tail = prev_id;
1148 }
1149
1150 list.size -= 1;
1151 return next_id;
1152 }

1267 vertex_iterator erase(vertex_iterator v) {
1268 if (v == end_vertices()) {
1269 return end_vertices();
1270 }
1271 for (EdgeDir dir : {EdgeDir::Incoming, EdgeDir::Outgoing}) {
1272 for (auto e = begin_incident_edges(v, dir); e != end_edges(); ) {
1273 // remove the edge from the other vertex's list
1274 const Edge& this_edge = e->edge();
1275 // if we're iterating over this vertex's incoming edges, then this vertex is
1276 // the destination, and the other vertex is the edge's source. Vice versa for
1277 // the other direction.
1278 VertexId other_id = (dir == EdgeDir::Incoming) ? this_edge.v0 : this_edge.v1;
1279 vertex_iterator other = find_vertex(other_id);
1280 // remove the edge from the other vertex's (opposite-direction) list
1281 _excise_from_list(other, ~dir, e.inner_iterator());
1282
1283 // don't bother removing the edge from the current list, because we're
1284 // about to erase it anyway. increment the iterator by following its link
1285 // to the next edge now; after we erase the edge, the iterator will
1286 // be invalid.

1289 // erase the edge
1290 _edges.erase(e_iter);
1291 if (not next_id) break;
1292 e = incident_edge_iterator {this, _edges.find(*next_id), dir, v.id()};
1293 }
1294 }
1295 return {this, _verts.erase(v.inner_iterator())};

1358 void _splice_edge(
1359 EdgeNode& edge,
1360 VertexNode& shared_vert,
1361 edge_iterator before,
1362 EdgeDir dir,
1363 EdgeId eid)
1364 {
1365 EdgeLink& link = edge.links[(int) dir];
1366 if (before != end_edges()) {
1367 // inserting before an existing edge
1368 EdgeLink& next_link = before.node().links[(int) dir];
1369 link.next = before.id();
1370 link.prev = next_link.prev;
1371 next_link.prev = eid;
1372 } else {
1373 // appending to the end of the list
1374 EdgeList& edge_list = shared_vert.edges[(int) dir];
1375 link.next = std::nullopt;
1376 link.prev = edge_list.tail;
1377 edge_list.tail = eid; // incoming edge is the new tail
1378 }
1379 if (not link.prev) {
1380 // we're inserting a new list head
1381 shared_vert.edges[(int) dir].head = eid;
1382 }

1541 bool swap_edge_order(edge_iterator e0, edge_iterator e1, EdgeDir dir) {
1542 if (e0->vertex(dir) != e1->vertex(dir)) return false;
1543 if (e0 == e1) return true;
1544
1545 int idx = (int) dir;
1546 EdgeLink e0link = e0.node().links[idx];
1547 EdgeLink e1link = e1.node().links[idx];
1548
1549 // e0p <-> [e0] <-> e0n ... e1p <-> [e1] <-> e1n
1550 // e0p <-> [e1] <-> e0n ... e1p <-> [e0] <-> e1n
1551
1552 // repair the previous and next nodes
1553 if (e0link.prev) _edges.find(*e0link.prev)->second.links[idx].next = e1->id();
1554 if (e0link.next) _edges.find(*e0link.next)->second.links[idx].prev = e1->id();
1555 if (e1link.prev) _edges.find(*e1link.prev)->second.links[idx].next = e0->id();
1556 if (e1link.next) _edges.find(*e1link.next)->second.links[idx].prev = e0->id();
1557
1558 // handle the head of the list
1559 if (not (e0link.prev and e1link.prev)) {
1560 // one of the nodes is the head of the list.
1561 // we need to fix up that vertex to point to the new head.
1562 if (e0link.prev) {
1563 // e1 is the current head of the list
1564 // ensure e0 is the current head
1565 std::swap(e0, e1);
1566 }
1567 // e1 will be the new head of the list
1568 VertexId vid = e1->vertex(dir);
1569 VertexNode& v = _verts.find(vid)->second;
1570 v.edges[idx].head = e1->id();
1571 }
1572
1573 // handle the tail of the list
1574 if (not (e0link.next and e1link.next)) {
1575 // one of the nodes is the tail of the list.
1576 // we need to fix up that vertex to point to the new tail.
1577 if (e0link.next) {
1578 // e1 is the current tail of the list
1579 // ensure e0 is the current tail
1580 std::swap(e0, e1);
1581 }
1582 // e1 will be the new tail of the list
1583 VertexId vid = e1->vertex(dir);
1584 VertexNode& v = _verts.find(vid)->second;
1585 v.edges[idx].tail = e1->id();
1586 }
1587
1588 // exchange the links
1589 std::swap(e0.node().links[idx], e1.node().links[idx]);
1590
1591 return true;
1592 }
